function isPrime(number) {
  if (number==0 || number==1) {
    return false
  }
  for (let i = 2; i <= number / i ; i++) {
		if (number % i == 0) {
			return false
		}
	}
  return true
}

function FindAllPrime(number) {

	let allPrime = []

	for (let i = 0; i < number; i++) {
		if (isPrime(i)){
			allPrime.push(i)
		}
	}

	return allPrime
}

	let allPrime = FindAllPrime(number)

}
